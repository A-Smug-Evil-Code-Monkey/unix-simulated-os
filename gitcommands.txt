//initialize git repo
git init

//add remote repo location
git remote add origin https://github.com/joshriccio/CSC452_USLOSS.git

//pull the master code from the online repo
git pull origin master 

//check the status
git status

//Add all c files, or use *.h for header files etc.
git add *.c

//make a commit to your local repo
git commit -am "first commit, updating all files"

//Push changes to the branch, preferably create a branch and push to branch
git push origin <branchName>
